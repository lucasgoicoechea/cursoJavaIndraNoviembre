CREATE TABLE `coches` ( `id` int NOT NULL, `marca` varchar(255) DEFAULT NULL, `anio` int NOT NULL, `rueda_id` int DEFAULT NULL, PRIMARY KEY (`id`), KEY `FK_nombre` (`rueda_id`) ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;
CREATE TABLE `ruedas` ( `id` int NOT NULL AUTO_INCREMENT, `marca` varchar(45) DEFAULT NULL, PRIMARY KEY (`id`)) ENGINE=InnoDB AUTO_INCREMENT=2 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;
